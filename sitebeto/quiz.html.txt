    <script>
        function checkQuiz() {
            const answers = {
                q1: "Azul",
                q2: "Jogos",
                q3: "Pizza"
            };

            let score = 0;

            for (let q in answers) {
                const selected = document.querySelector(`input[name="${q}"]:checked`);
                if (selected && selected.value === answers[q]) {
                    score++;
                }
            }

            document.getElementById("result").innerText = `Você acertou ${score} de 3 perguntas!`;
        }
    </script>
